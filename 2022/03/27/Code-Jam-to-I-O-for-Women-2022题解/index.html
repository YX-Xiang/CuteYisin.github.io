struct leaf
{
    int b,e,num;
    friend bool operator < (leaf a, leaf b)
    {
        return a.e > b.e;
    }
};

leaf y[105];
priority_queue<leaf>q,qq;

int main()
{
    ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
    int t,cc=1;cin>>t;
    while(t--)
    {
        int d,n,u,ans=0;cin>>d>>n>>u;
        q=qq;
        for(int i=0;i<d;i++)
        {
            int m,l,e;
            cin>>m>>l>>e;
            y[i].b=m;
            y[i].e=m+e-1;
            y[i].num=l;
            q.push(y[i]);
        }
        int f=1;
        for(int i=0;i<n;i++)
        {
            int o,nn=u,ff=0;cin>>o;
            vector<leaf>v;
            while(!q.empty()&&f)
            {
                leaf x=q.top();
                //cout<<x.b<<" "<<x.e<<" "<<x.num<<endl;
                if(x.e<o)
                {
                    q.pop();
                    continue;
                }
                else if(x.b>o)
                {
                    q.pop();
                    v.push_back(x);
                    continue;
                }
                else if(x.num>nn)
                {
                    x.num-=nn;
                    q.pop();
                    q.push(x);
                    ff=1;
                    break;
                }
                else if(x.num==nn)
                {
                    q.pop();
                    ff=1;
                    break;
                }
                else
                {
                    q.pop();
                    nn-=x.num;
                    continue;
                }
            }
            if(!ff) f=0;
            else ans++;
            for(int i=0;i<v.size();i++) q.push(v[i]);
        }
        cout<<"Case #"<<cc++<<": "<<ans<<endl;
    }
    return 0;
}
