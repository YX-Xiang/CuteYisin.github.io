    int n;

    for(int i=1;i<=n;i++)

    for(int i=1;i<=n;i++)
    {
        while(!s.empty()&&a[s.top()]>a[i]) s.pop();
        if(s.empty()) l[i]=1ll*i*a[i];
        else l[i]=l[s.top()]+1ll*(i-s.top())*a[i];

    {
        while(!s.empty()&&a[s.top()]>a[i]) s.pop();

        s.push(i);

    ll ans=0;
    int p;
    for(int i=1;i<=n;i++)

        {
            ans=l[i]+r[i]-a[i];
            p=i;

            a[i]=a[i-1];
